CSS Cheat Sheet

Selectors

/* Basic Selectors */
.class-name     /* Class selector */
#id-name        /* ID selector */
div             /* Element selector */
div p           /* Descendant selector */
div > p         /* Child selector */
div + p         /* Adjacent sibling */
div ~ p         /* General sibling */
[attr="value"]  /* Attribute selector */

/* Pseudo-classes */
:hover          /* Mouse over */
:focus          /* Element has focus */
:first-child    /* First child element */
:last-child     /* Last child element */
:nth-child(n)   /* nth child element */

/* Pseudo-elements */
::before        /* Before element */
::after         /* After element */
::first-line    /* First line */
::first-letter  /* First letter */

Box Model

.box {
/* Content dimensions */
    width: 200px;
    height: 100px;
/* Padding */
    padding: 10px;             /* All sides */
    padding: 10px 20px;        /* Top/bottom left/right */
    padding: 10px 20px 15px;   /* Top left/right bottom */
    padding: 10px 20px 15px 25px; /* Top right bottom left */
/* Border */
    border: 1px solid black;
    border-radius: 5px;
/* Margin */
    margin: 10px;
/* Box sizing */
    box-sizing: border-box;
    }

Flexbox

.container { 
    display: flex;
/* Main axis */
    justify-content: flex-start | center | flex-end | space-between | space-around;
/* Cross axis */
    align-items: stretch | flex-start | center | flex-end;
/* Multiple lines */
    flex-wrap: nowrap | wrap | wrap-reverse;
/* Direction */
    flex-direction: row | column | row-reverse | column-reverse;
    }
.item { 
    flex: 1;                    /* Flex grow */
    flex: 1 1 auto;            /* grow shrink basis */
    align-self: center;        /* Individual alignment */
    order: 1;                  /* Order in flex container */
    }

Grid

.grid-container {
    display: grid;
/* Define columns/rows */
    grid-template-columns: repeat(3, 1fr);
    grid-template-rows: auto;
/* Gaps */
    gap: 20px;
/* Areas */
    grid-template-areas: 
        "header header"
        "sidebar main"
        "footer footer";
        }
.grid-item {
    grid-column: 1 / 3;        /* Column start/end */
    grid-row: 1 / 2;          /* Row start/end */
    grid-area: header;        /* Named area */
    }

Positioning

.element {
    position: static | relative | absolute | fixed | sticky;
    top: 0;
    right: 0;
    bottom: 0;
    left: 0;
    z-index: 1;
/* Float */
    float: left | right | none;
    clear: both;
/* Display */    
    display: block | inline | inline-block | none;
    }

Animations & Transitions

/* Transition */
.element {
    transition: all 0.3s ease;
    transition: property duration timing-function delay;
    }
/* Animation */
@keyframes slidein {
    from {
        transform: translateX(0);
    }
    to {
        transform: translateX(100%);
    }
}
.animated {
    animation : slidein 1s ease infinite;
    animation : name duration timing-function delay iteration-count direction fill-mode;
    }
