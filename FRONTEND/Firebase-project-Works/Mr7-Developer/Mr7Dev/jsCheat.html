JavaScript Cheat Sheet

Variables & Data Types

// Variable declaration
let variable = 'value';
const constant = 'unchangeable';
var oldWay = 'not recommended';

// Data Types
let string = 'text';
let number = 42;
let boolean = true;
let array = [1, 2, 3];
let object = { key: 'value' };
let nullValue = null;
let undefined;

Functions

// Regular function
function add(a, b) {
    return a + b;
}

// Arrow function
const multiply = (a, b) => a * b;

// Function expression
const greet = function(name) {
    return `Hello, ${name}!`;
};

Arrays

const fruits = ['apple', 'banana', 'orange'];

// Array methods
fruits.push('grape');     // Add to end
fruits.pop();            // Remove from end
fruits.unshift('lemon'); // Add to start
fruits.shift();          // Remove from start
fruits.slice(1, 3);      // Get subset
fruits.splice(1, 1);     // Remove elements

// Array iteration
fruits.forEach(fruit => console.log(fruit));
const newArray = fruits.map(fruit => fruit.toUpperCase());

Objects

const person = {
    name: 'John',
    age: 30,
    greet() {
        return `Hello, I'm ${this.name}`;
    }
};

// Object operations
Object.keys(person);   // Get all keys
Object.values(person); // Get all values
Object.entries(person); // Get key-value pairs

ES6+ Features

// Destructuring
const { name, age } = person;
const [first, second] = array;

// Spread operator
const newArray = [...oldArray];
const newObject = { ...oldObject };

// Template literals
const message = `Hello ${name}!`;

// Optional chaining
const value = object?.property?.subProperty;

// Nullish coalescing
const result = value ?? defaultValue;

Promises & Async

// Promise
const promise = new Promise((resolve, reject) => {
    // Async operation
    if (success) {
        resolve(result);
    } else {
        reject(error);
    }
});

// Async/Await
async function getData() {
    try {
        const response = await fetch(url);
        const data = await response.json();
        return data;
    } catch (error) {
        console.error(error);
    }
}

DOM Manipulation

// Selecting elements
const element = document.querySelector('.class');
const elements = document.querySelectorAll('.class');

// Modifying elements
element.innerHTML = 'New content';
element.classList.add('new-class');
element.style.color = 'red';

// Event handling
element.addEventListener('click', (e) => {
    e.preventDefault();
    // Handle event
});
